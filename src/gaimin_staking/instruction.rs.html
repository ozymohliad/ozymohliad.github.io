use arrayref::{array_ref, array_refs};
use shank::ShankContext;
use solana_program::{clock::UnixTimestamp, program_error::ProgramError, msg};
use std::{mem, fmt::Debug};

use crate::{error::GaiminError, utils::parse_string, state::BNB_CHAIN_WALLET_ADDRESS_LENGTH};

/// Instructions supported by the GMRX Claim Program
#[derive(Debug, ShankContext)]
pub enum GaiminInstruction {
    /// Instruction code: `0x0`
    ///
    /// Create and initialize the config account. It is a system instruction that can be executed
    /// by anyone. The config account may NOT be initialized. The `authority` account will pay for
    /// rent exemption and become the config authority with the exclusive right to execute other
    /// system instructions. To update the config account, it must first be deleted and then
    /// created again.
    #[account(0, signer, name = "authority", desc = "Future config authority/Fee payer")]
    #[account(1, writable, name = "config", desc = "PDA that the config account will be created at")]
    #[account(2, name = "system_program", desc = "System program")]
    Config(ConfigArgs),

    /// Instruction code: `0x1`
    ///
    /// Delete an account owned by this program. It is a system instruction that must be signed by
    /// the config authority. The config account must be initialized first. The account to be
    /// deleted must also be initialized.
    #[account(0, signer, name = "authority", desc = "Config authority")]
    #[account(1, writable, name = "target", desc = "Account to be deleted")]
    #[account(2, name = "config", desc = "Config account PDA")]
    Delete,

    /// Instruction code: `0x2`
    ///
    /// Create and initialize an NFT record account. It is a system instruction that must be signed
    /// by the config authority. The config account must be initialized first. The NFT record
    /// account may NOT be initialized. To update an existing NFT record, delete it and create
    /// again. The NFT being registered must be valid and of the programmable standard. The config
    /// authority will pay for rent exemption.
    #[account(0, signer, name = "authority", desc = "Config authority/Fee payer")]
    #[account(1, name = "mint", desc = "Mint address of the NFT")]
    #[account(2, name = "metadata", desc = "Metadata address of the NFT")]
    #[account(3, name = "edition", desc = "Edition account address of the NFT")]
    #[account(4, writable, name = "nft_record", desc = "PDA that the NFT record account will be created at")]
    #[account(5, name = "config", desc = "Config account PDA")]
    #[account(6, name = "system_program", desc = "System program")]
    Nft,

    /// Instruction code: `0x3`
    ///
    /// Create and initialize the user's receiver record account. It is a user instruction and
    /// must be signed with the user's wallet account key. The receiver account may or may not be
    /// initialized. If the account is initialized, its data will be overwritten. The user will pay
    /// for rent exemption.
    #[account(0, signer, name = "user", desc = "User wallet account/Fee payer")]
    #[account(1, writable, name = "receiver", desc = "PDA that the receiver account will be created at")]
    #[account(2, name = "system_program", desc = "System program")]
    Receiver(ReceiverArgs),

    /// Instruction code: `0x4`
    ///
    /// Create and initialize a claim record for the given user and NFT. It is a user instruction
    /// and must be signed with the user's wallet account key. The user must own the NFT and the
    /// token account must be locked. The claim record account may NOT be initialized. The config
    /// account, the NFT record account, and the receiver record account must be initialized. The
    /// user will pay for rent exemption.
    #[account(0, signer, name = "user", desc = "User wallet account that owns the NFT/Fee payer")]
    #[account(1, name = "mint", desc = "Mint address of the NFT")]
    #[account(2, name = "token", desc = "User's token account")]
    #[account(3, name = "token_record", desc = "User's token record account (for locking validation)")]
    #[account(4, writable, name = "nft_record", desc = "NFT record account PDA")]
    #[account(5, writable, name = "claim", desc = "PDA that the claim account will be created at")]
    #[account(6, name = "config", desc = "Config account PDA")]
    #[account(7, name = "receiver", desc = "PDA of the receiver record")]
    #[account(8, name = "system_program", desc = "System program")]
    Claim,
}

impl GaiminInstruction {
    pub fn unpack(input: &[u8]) -> Result<Self, ProgramError> {
        let (tag, rest) = input
            .split_first()
            .ok_or(ProgramError::InvalidInstructionData)?;

        Ok(match tag {
            0 => Self::Config(ConfigArgs::unpack(rest)?),
            1 => Self::Delete,
            2 => Self::Nft,
            3 => Self::Receiver(ReceiverArgs::unpack(rest)?),
            4 => Self::Claim,
            i => {
                msg!("[Error] Invalid instruction code: {}", i);
                return Err(GaiminError::InvalidInstruction.into())
            }
        })
    }
}

#[derive(Debug)]
pub struct ConfigArgs {
    pub claimable_from: UnixTimestamp,
    pub total_reward: f64,
    pub initial_reward_frac: f32,
    pub reward_period_sec: i32,
}

impl ConfigArgs {
    fn unpack(input: &[u8]) -> Result<Self, ProgramError> {
        let src = array_ref![
            input,
            0,
            mem::size_of::<UnixTimestamp>()
                + mem::size_of::<f64>()
                + mem::size_of::<f32>()
                + mem::size_of::<i32>()
        ];
        let (claimable_from, total_reward, initial_reward_frac, reward_period_sec) = array_refs![
            src,
            mem::size_of::<UnixTimestamp>(),
            mem::size_of::<f64>(),
            mem::size_of::<f32>(),
            mem::size_of::<i32>()
        ];

        Ok(Self {
            claimable_from: i64::from_le_bytes(*claimable_from),
            total_reward: f64::from_le_bytes(*total_reward),
            initial_reward_frac: f32::from_le_bytes(*initial_reward_frac),
            reward_period_sec: i32::from_le_bytes(*reward_period_sec),
        })
    }
}

#[derive(Debug)]
pub struct ReceiverArgs {
    pub bnb_chain_wallet_address: String,
}

impl ReceiverArgs {
    fn unpack(input: &[u8]) -> Result<Self, ProgramError> {
        parse_string(input, BNB_CHAIN_WALLET_ADDRESS_LENGTH)
            .map(|bnb_chain_wallet_address| Self { bnb_chain_wallet_address })
    }
}
